\begin{recipe}{Fish Tacos}{}{}	
	\ingredient[180]{g}{All-Purpose Flour}
	\ingredient[1]{g}{Baking Powder}
	\ingredient[10]{g}{Salt}
	\ingredient[10]{g}{Black Pepper}
	\ingredient[15]{g}{Garlic Powder}
	\ingredient[15]{g}{Onion Powder}
	\ingredient[10]{g}{Paprika}
	Combine until homogenous.
	\ingredient[12]{fl. oz.}{Pale Lager Beer}
	Slowly pour in beer, whisking until very smooth
	\ingredient[1]{lb}{Cod or Halibut}
	Cut into one-inch thick strips. 
	\ingredient[30]{g}{All-Purpose Flour}
	\ingredient[4]{Cups}{Frying Oil}
	Heat oil to 375 degrees. 
	Dust fish with flour, then dip in beer batter. With tongs, slowly lower battered fish in oil, releasing when it starts bubbling. Deep-fry until golden brown and cooked through (145 Â°F). 
	\ingredient[]{}{Corn tortillas}
	Heat tortillas on a non-stick pan. 
	\ingredient[]{}{Cole-Slaw Mix}
	\ingredient[1]{}{Avocado}
	\ingredient[]{}{Siracha Mayo}
	Place fish on tortilla. Top with cole-slaw mix and a slice of avocado. Drizzle with Siracha mayo. 
\end{recipe}
