	\ingredient[2]{Tsp}{Garlic, Minced}
	\ingredient[30]{g}{Lemongrass, Minced}
	\ingredient[30]{g}{Water}
	\ingredient[45]{g}{Sugar}
	\ingredient[45]{g}{Fish Sauce}
	\ingredient[10]{g}{Black Pepper}
	Blend all ingredients to make a marinade. 
	\ingredient[1]{lb}{Pork or Beef}
	Slice meat into thin strips and Marinade for about an hour. 
	\ingredient[1]{Tbsp}{Oil}
	Add oil to a wok, and heat until smoking. Saut√© the pork until cooked thoroughly. 
\end{recipe}
