\begin{recipe}{Pork-Taro Spring Roll}{}{Makes About 50 small egg rolls.}
	\ingredient[1]{lb}{Taro}
	\ingredient[1]{lb}{Carrot}
	Shred carrots and taro, then squeeze out any excess moisture with a paper towel. 
	\ingredient[1]{large}{Egg}
	Separate and set whites aside. 
	\ingredient[1]{tbsp}{Salt}
	\ingredient[1]{tbsp}{Ground Black Pepper}
	\ingredient[1]{tsp}{Ginger Powder}
	\ingredient[1]{tsp}{Garlic Powder}
	Mix the dry seasonings in a small bowl until uniform. 
	\ingredient[1]{lb}{Ground Pork}
	\ingredient[1]{tsp}{Toasted Sesame Oil}
	Sprinkle the seasoning mix over the ground pork and work the spices into the meat in a large bowl. Then combine with vegetables.
	\ingredient[50]{}{TYJ Spring Roll Pastries} 
	Roll spring rolls (use egg whites to seal the edges) and ar-fry in batches at $390^\circ$F until golden brown (these freeze well). Use a food thermometer to confirm that internals are $165^\circ F$.
	\ingredient[]{}{}
\end{recipe}
