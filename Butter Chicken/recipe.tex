\begin{recipe}{Honey Walnut Shrimp}{}{}	
	\ingredient[240]{g}{Yogurt}
	\ingredient[10]{g}{Garam Masala (aka Curry Powder)}
	\ingredient[5]{g}{Cumin}
	\ingredient[5]{g}{Garlic Powder}
	\ingredient[5]{g}{Lime Juice}
	\ingredient[15]{g}{Salt}
	\ingredient[1.5]{lb}{Boneless Chicken Thighs}
	Whisk together yogurt and seasonings. Marinade the chicken in the mixture for 30 minutes and sear in batches. Then set aside. 
	
	\ingredient[1]{}{Yellow Onion, Diced}
	\ingredient[10]{g}{Salt}
	\ingredient[10]{g}{Pepper}
	Lower pan to medium heat and saute until softened. 
	\ingredient[5]{}{g}{Garlic Powder}
	\ingredient[5]{}{g}{Onion Powder}
	\ingredient[15]{}{g}{Garam Masala}
	\ingredient[10]{}{g}{Paprika}
	\ingredient[2]{tsp}{ground cumin}
	\ingredient[6]{}{g}{Tumeric}
	\ingredient[100]{g}{Crushed cashews}
	Add spices and season until spices have toasted.
	\ingredient[14]{oz}{Crushed Tomatoes}
	\ingredient[6]{oz}{water}
	Add in crushed tomatoes and water and simmer until sauce has been reduced. Add in chicken and simmer an additional 5 minutes.
	\ingredient[1]{can}{Evaporated Milk}
	Stir in evaporated milk until thickened and incorporated. Simmer an additional 3-4 minutes.
	\ingredient[2]{Tbps}{Unsalted butter}
	\ingredient[]{}{Salt}
	Turn off the heat and stir in unsalted butter until emulsified. Season to taste with salt. Serve with rice and cilantro.
\end{recipe}
