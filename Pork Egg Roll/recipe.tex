\begin{recipe}{Pork-Taro Spring Roll}{}{Makes About 50 small egg rolls.}
	\ingredient[0.75]{lb}{Taro}
	\ingredient[0.75]{lb}{Carrot}
	Shred carrots and taro; wring out excess moisture using a clean cheesecloth.
	\textbf{Note:} Raw taro contains calcium oxalate crystals which can cause skin irritation for some people. Be sure to wear gloves when shredding it
	\ingredient[2]{large}{Eggs}
	Separate eggs, setting whites aside.
	\ingredient[15]{g}{Salt}
	\ingredient[5]{g}{MSG}
	\ingredient[5]{g}{Ground Black Pepper}
	\ingredient[3]{g}{Ginger Powder}
	\ingredient[5]{g}{Garlic Powder}
	Mix the dry seasonings in a small bowl until uniform. 
	\ingredient[0.5]{lb}{Shrimp}
	Chop the shrimp into small pieces and pat them dry with a paper towel.
	\ingredient[1]{lb}{Ground Pork}
	\ingredient[5]{g}{Toasted Sesame Oil}
	In a large bowl, work seasonings and sesame oil into the ground pork and chopped shrimp until the meat becomes "tacky." Fold in the shredded vegetables and egg yolks.
	\ingredient[50]{}{TYJ Spring Roll Pastries} 
	Keep TYJ pastries under a damp cloth while working. Place 1-2 tbsp of filling per roll, wrap tightly, and seal edges with egg whites. Freeze in a single layer in ziplock bags (10 should fit in a quart bag) until ready to cook.
	To cook: air fry at $350^\circ$F for 5 minutes to thaw. Lightly spray with oil, then increase heat to $390^\circ\text{F}$, turning halfway, until golden brown and internal temp reaches $165^\circ\text{F}$.
\end{recipe}
